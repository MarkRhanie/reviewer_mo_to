class AuthService extends ChangeNotifier {
  Map<String, dynamic>? _currentUser;
  bool get isLoggedIn => _currentUser != null;
  Map<String, dynamic>? get currentUser => _currentUser;

  // UPDATE PROFILE - Supports both mobile (file path) and web (base64)
  Future<void> updateProfile({
    required String displayName,
    required String bio,
    required String website,
    required String location,
    String? profileImagePath,   // For mobile
    String? profileImageBase64, // For web
  }) async {
    try {
      if (_currentUser == null) return;

      final users = await getAllUsers();
      final userIndex = users.indexWhere((u) => u['id'] == _currentUser!['id']);
      
      if (userIndex != -1) {
        // Update text fields
        users[userIndex]['fullName'] = displayName;
        users[userIndex]['bio'] = bio;
        users[userIndex]['website'] = website;
        users[userIndex]['location'] = location;

        // Save image path for mobile
        if (profileImagePath != null) {
          users[userIndex]['profileImagePath'] = profileImagePath;
        }

        // Save base64 for web
        if (profileImageBase64 != null) {
          users[userIndex]['profileImageBase64'] = profileImageBase64;
        }

        await _saveUsers(users);
        _currentUser = users[userIndex];
        await _saveCurrentSession(_currentUser!);
        notifyListeners();
      }
    } catch (e) {
      print('Error updating profile: $e');
    }
  }

  // GET ALL REGISTERED USERS
  Future<List<Map<String, dynamic>>> getAllUsers() async {
    final prefs = await SharedPreferences.getInstance();
    final usersJson = prefs.getString(_usersKey);
    if (usersJson != null) {
      return List<Map<String, dynamic>>.from(json.decode(usersJson));
    }
    return [];
  }

  // REGISTER NEW USER WITH DEFAULT PROFILE FIELDS
  Future<String?> register({
    required String username,
    required String email,
    required String password,
    String? fullName,
  }) async {
    try {
      final users = await getAllUsers();

      // Check for existing user
      if (users.any((user) => user['email'] == email)) {
        return 'Email already exists';
      }
      if (users.any((user) => user['username'] == username)) {
        return 'Username already taken';
      }

      // Create new user with default profile values
      final newUser = {
        'id': DateTime.now().millisecondsSinceEpoch,
        'username': username,
        'email': email,
        'password': password,
        'fullName': fullName ?? username.replaceFirst('@', ''),
        'bio': '',           // Empty bio by default
        'website': '',       // Empty website
        'location': '',      // Empty location
        'profileImagePath': null,    // For mobile
        'profileImageBase64': null,  // For web
        'createdAt': DateTime.now().toIso8601String(),
      };

      users.add(newUser);
      await _saveUsers(users);
      return null;
    } catch (e) {
      return 'An error occurred: $e';
    }
  }

  // LOGOUT
  Future<void> logout() async {
    try {
      final prefs = await SharedPreferences.getInstance();
      await prefs.remove(_currentUserKey);
      _currentUser = null;
      notifyListeners();
    } catch (e) {
      print('Error logging out: $e');
    }
  }
}

class EditProfileScreen extends StatefulWidget {
  const EditProfileScreen({super.key});

  @override
  State<EditProfileScreen> createState() => _EditProfileScreenState();
}

class _EditProfileScreenState extends State<EditProfileScreen> {
  final _formKey = GlobalKey<FormState>();
  final _displayNameController = TextEditingController();
  final _bioController = TextEditingController();
  final _websiteController = TextEditingController();
  final _locationController = TextEditingController();

  // Image handling - different for mobile and web
  File? _profileImageFile;           // Mobile
  String? _profileImageBase64;       // Web
  Uint8List? _profileImageBytes;     // Web

  final ImagePicker _picker = ImagePicker();

  @override
  void initState() {
    super.initState();
    _loadUserData();
  }

  // LOAD EXISTING PROFILE DATA
  void _loadUserData() {
    final authService = Provider.of<AuthService>(context, listen: false);
    final user = authService.currentUser;

    _displayNameController.text = user?['fullName'] ?? '';
    _bioController.text = user?['bio'] ?? '';
    _websiteController.text = user?['website'] ?? '';
    _locationController.text = user?['location'] ?? '';
    
    _loadSavedImage();
  }

  // LOAD SAVED PROFILE IMAGE
  Future<void> _loadSavedImage() async {
    final authService = Provider.of<AuthService>(context, listen: false);
    final user = authService.currentUser;

    // For Web: Load base64 image
    if (kIsWeb) {
      final imageBase64 = user?['profileImageBase64'];
      if (imageBase64 != null && imageBase64.isNotEmpty) {
        try {
          final bytes = base64.decode(imageBase64);
          setState(() {
            _profileImageBytes = bytes;
            _profileImageBase64 = imageBase64;
          });
        } catch (e) {
          print('Error decoding image: $e');
        }
      }
    }
    // For Mobile: Load file path
    else {
      final imagePath = user?['profileImagePath'];
      if (imagePath != null && imagePath.isNotEmpty) {
        try {
          final file = File(imagePath);
          if (await file.exists()) {
            setState(() {
              _profileImageFile = file;
            });
          }
        } catch (e) {
          print('Error loading file: $e');
        }
      }
    }
  }

  // PICK IMAGE FROM GALLERY
  Future<void> _pickImageFromGallery() async {
    // For mobile Android, request permission
    if (!kIsWeb && Theme.of(context).platform == TargetPlatform.android) {
      bool hasPermission = await _requestPermission(Permission.photos);
      if (!hasPermission) {
        _showSnackBar('Storage permission denied', isError: true);
        return;
      }
    }

    try {
      final XFile? image = await _picker.pickImage(
        source: ImageSource.gallery,
        maxWidth: 500,
        maxHeight: 500,
        imageQuality: 85,
      );

      if (image != null) {
        await _processPickedImage(image);
        _showSnackBar('Image selected successfully!');
      }
    } catch (e) {
      print('Error selecting image: $e');
      _showSnackBar('Error selecting image: $e', isError: true);
    }
  }

  // PROCESS PICKED IMAGE (Handle both mobile and web)
  Future<void> _processPickedImage(XFile image) async {
    final bytes = await image.readAsBytes();

    setState(() {
      if (kIsWeb) {
        _profileImageBytes = bytes;
        _profileImageBase64 = base64.encode(bytes);
        _profileImageFile = null;
      } else {
        _profileImageFile = File(image.path);
        _profileImageBytes = null;
        _profileImageBase64 = null;
      }
    });
  }

  // SAVE PROFILE CHANGES
  Future<void> _handleSave() async {
    if (!_formKey.currentState!.validate()) return;

    setState(() => _isLoading = true);

    final authService = Provider.of<AuthService>(context, listen: false);

    // For Web: Save base64
    if (kIsWeb) {
      await authService.updateProfile(
        displayName: _displayNameController.text,
        bio: _bioController.text,
        website: _websiteController.text,
        location: _locationController.text,
        profileImageBase64: _profileImageBase64,
        profileImagePath: null,
      );
    }
    // For Mobile: Save file path
    else {
      await authService.updateProfile(
        displayName: _displayNameController.text,
        bio: _bioController.text,
        website: _websiteController.text,
        location: _locationController.text,
        profileImagePath: _profileImageFile?.path,
        profileImageBase64: null,
      );
    }

    if (mounted) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
          content: const Text('Your profile has been updated!'),
          backgroundColor: AppColors.spotifyGreen,
          behavior: SnackBarBehavior.floating,
        ),
      );
      Navigator.pop(context, true);
    }
  }

  @override
  Widget build(BuildContext context) {
    final authService = Provider.of<AuthService>(context);
    final user = authService.currentUser;
    final rawUsername = user?['username'] ?? 'user';
    final displayUsername = rawUsername.startsWith('@') ? rawUsername : '@$rawUsername';

    bool hasImage = kIsWeb ? _profileImageBytes != null : _profileImageFile != null;

    return Scaffold(
      backgroundColor: AppColors.background,
      appBar: AppBar(
        backgroundColor: AppColors.card,
        leading: IconButton(
          icon: const Icon(Icons.close, color: Colors.white),
          onPressed: () => Navigator.pop(context),
        ),
        title: const Text('Edit Profile', style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold)),
        actions: [
          TextButton(
            onPressed: _isLoading ? null : _handleSave,
            child: _isLoading
                ? const CircularProgressIndicator(strokeWidth: 2, color: AppColors.spotifyGreen)
                : Text('Save', style: TextStyle(color: AppColors.spotifyGreen, fontSize: 16, fontWeight: FontWeight.bold)),
          ),
        ],
      ),
      body: SingleChildScrollView(
        padding: const EdgeInsets.all(20),
        child: Form(
          key: _formKey,
          child: Column(
            children: [
              // Profile Picture Section
              Center(

                  children: [
                    GestureDetector(
                      onTap: _showImagePicker,
                      child: Container(
                        decoration: BoxDecoration(
                          shape: BoxShape.circle,
                          border: Border.all(color: AppColors.spotifyGreen, width: 3),
                        ),
                        child: CircleAvatar(
                          radius: 60,
                          backgroundColor: Colors.pink,
                          backgroundImage: kIsWeb
                              ? (_profileImageBytes != null ? MemoryImage(_profileImageBytes!) : null)
                              : (_profileImageFile != null ? FileImage(_profileImageFile!) : null),
                          child: !hasImage
                              ? Text(
                                  displayUsername.length > 1 ? displayUsername[1].toUpperCase() : 'U',
                                  style: const TextStyle(color: Colors.white, fontSize: 40, fontWeight: FontWeight.bold),
                                )
                              : null,
                        ),
                      ),
                    ),
                    Positioned(
                      bottom: 0,
                      right: 0,
                      child: GestureDetector(

                          padding: const EdgeInsets.all(8),

  Widget _buildTextField({
    required TextEditingController controller,
    required String label,
    required IconData icon,
    required String hint,
    int maxLines = 1,
    bool required = true,
  }) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Text(label + (required ? ' *' : ''), style: TextStyle(color: Colors.grey[400], fontSize: 12, fontWeight: FontWeight.bold)),
        const SizedBox(height: 8),
        TextFormField(
          controller: controller,
          style: const TextStyle(color: Colors.white),
          maxLines: maxLines,
          decoration: InputDecoration(
            hintText: hint,
            hintStyle: const TextStyle(color: Colors.grey),
            prefixIcon: Icon(icon, color: Colors.grey),
            filled: true,
            fillColor: AppColors.card,
            border: OutlineInputBorder(borderRadius: BorderRadius.circular(15), borderSide: BorderSide.none),
            focusedBorder: OutlineInputBorder(
              borderRadius: BorderRadius.circular(15),
              borderSide: BorderSide(color: AppColors.spotifyGreen, width: 2),
            ),
          ),
          validator: (value) {
            if (required && (value == null || value.isEmpty)) {
              return 'Please enter your $label';
            }
            return null;
          },
        ),
      ],
    );
  }

    final prefs = await SharedPreferences.getInstance();

        _privateAccountEnabled = prefs.getBool('private_account') ?? false;

  @override
  Widget build(BuildContext context) {
    return Consumer2<AuthService, FollowService>(
      builder: (context, authService, followService, child) {
        final user = authService.currentUser;
        final String rawUsername = user?['username'] ?? 'spotifyuser';
        final String displayUsername = rawUsername.startsWith('@') ? rawUsername : '@$rawUsername';
        final displayName = user?['fullName'] ?? rawUsername.replaceFirst('@', '');
        final bio = user?['bio'] ?? '';

        // Handle profile image (mobile vs web)
        final profileImagePath = user?['profileImagePath'];
        final profileImageBase64 = user?['profileImageBase64'];

        ImageProvider? profileImage;
        if (profileImageBase64 != null && profileImageBase64.isNotEmpty) {
          try {
            final bytes = base64.decode(profileImageBase64);
            profileImage = MemoryImage(bytes);
          } catch (e) {
            debugPrint('Error decoding profile image: $e');
          }

            final file = File(profileImagePath);

        if (post is PostModel) {
          return _buildMusicPostCard(post);
        } else if (post is ThreadPost) {
          return Padding(padding: const EdgeInsets.only(bottom: 12), child: ThreadCard(post: post));
        }
        return const SizedBox.shrink();
      },
    );
  }

  // SHOW FOLLOWERS LIST
  Future<void> _showFollowersList() async {
    final followService = Provider.of<FollowService>(context, listen: false);
    final List<Map<String, dynamic>> followers = [];
    for (String userId in followService.followersList) {
      final details = await followService.getUserDetails(userId);
      followers.add(details);
    }
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => FollowListScreen(
          title: 'Followers',
          users: followers,
          onFollowTap: (user) async {
            if (user['isFollowing']) {
              await followService.followUser(user['id'].toString());
            } else {
              await followService.unfollowUser(user['id'].toString());
            }
          },
        ),
      ),
    );
  }

  // SHOW FOLLOWING LIST
  Future<void> _showFollowingList() async {

    final List<Map<String, dynamic>> following = [];

          title: 'Following',

          onFollowTap: (user) async {
            if (user['isFollowing']) {
              await followService.followUser(user['id'].toString());
            } else {
              await followService.unfollowUser(user['id'].toString());
            }
          },
        ),
      ),
    );
  }

  // SHOW SETTINGS MENU (Bottom Sheet)
  void _showSettingsMenu() {
    showModalBottomSheet(
      context: context,
      backgroundColor: Colors.transparent,
      builder: (BuildContext context) {
        return Container(
          decoration: BoxDecoration(
            color: AppColors.card,
            borderRadius: const BorderRadius.vertical(top: Radius.circular(30)),
          ),
          child: Column(
            mainAxisSize: MainAxisSize.min,
            children: [
              Container(
                margin: const EdgeInsets.symmetric(vertical: 12),
                width: 40,
                height: 4,
                decoration: BoxDecoration(color: Colors.grey[600], borderRadius: BorderRadius.circular(2)),
              ),
              ListTile(
                leading: Container(
                  padding: const EdgeInsets.all(8),
                  decoration: BoxDecoration(color: AppColors.spotifyGreen.withValues(alpha: 0.1), shape: BoxShape.circle),
                  child: Icon(Icons.settings, color: AppColors.spotifyGreen),
                ),
                title: const Text('Settings', style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold)),
                subtitle: const Text('App preferences and account settings', style: TextStyle(color: Colors.grey)),
                onTap: () {
                  Navigator.pop(context);
                  Navigator.push(context, MaterialPageRoute(builder: (context) => const SettingsScreen()));
                },
              ),
              ListTile(
                leading: Container(
                  padding: const EdgeInsets.all(8),
                  decoration: BoxDecoration(color: Colors.red.withValues(alpha: 0.1), shape: BoxShape.circle),
                  child: const Icon(Icons.logout, color: Colors.red),
                ),
                title: const Text('Logout', style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold)),
                subtitle: const Text('Sign out of your account', style: TextStyle(color: Colors.grey)),
                onTap: () {
                  Navigator.pop(context);
                  _handleLogout();
                },
              ),
            ],
          ),
        );
      },
    );
  }

  // HANDLE LOGOUT
  Future<void> _handleLogout() async {
    final authService = Provider.of<AuthService>(context, listen: false);
    final bool? confirm = await showDialog(
      context: context,
      builder: (context) => AlertDialog(
        backgroundColor: AppColors.card,
        title: const Text('Logout', style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold)),
        content: const Text('Are you sure you want to logout?', style: TextStyle(color: Colors.grey)),
        actions: [
          TextButton(onPressed: () => Navigator.pop(context, false), child: Text('Cancel', style: TextStyle(color: Colors.grey[400]))),
          ElevatedButton(
            onPressed: () => Navigator.pop(context, true),
            style: ElevatedButton.styleFrom(backgroundColor: Colors.red),
            child: const Text('Logout'),
          ),
        ],
      ),
    );

    if (confirm == true && mounted) {
      await authService.logout();
      if (mounted) {
        Navigator.of(context).pushAndRemoveUntil(
          MaterialPageRoute(builder: (_) => const LoginScreen()),
          (route) => false,
        );
      }
    }
  }
}
